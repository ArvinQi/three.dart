dynamic update({num time}) #

update( {num time} ) {

 if ( _tweens.isEmpty ) { return false; }

 var i = 0, l = _tweens.length;

 if (time == null) {
   time = new DateTime.now().millisecondsSinceEpoch;
 }

 while ( i < l ) {
   if ( _tweens[ i ].update( time ) ) {
     i ++;
   } else {
     _tweens.removeAt( i );
     l--;
   }
 }
 return true;
}
